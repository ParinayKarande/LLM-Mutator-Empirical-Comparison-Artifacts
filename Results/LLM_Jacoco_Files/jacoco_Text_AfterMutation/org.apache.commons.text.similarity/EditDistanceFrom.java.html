package org.apache.commons.text.similarity;

import org.apache.commons.lang3.Validate;

public class EditDistanceFrom<R> {

    private final EditDistance<R> editDistance;

    private final CharSequence left;

    public EditDistanceFrom(final EditDistance<R> editDistance, final CharSequence left) {
        // Negate Conditionals: Changed != to ==
        Validate.isTrue(editDistance == null, "The edit distance may not be null."); 
        this.editDistance = editDistance;
        this.left = left;
    }

    // Return Values: Changing the return type from R to a different object type (e.g., String)
    // Just for the purpose of the mutation
    public R apply(final CharSequence right) {
        // Math: For demonstration, changing the behavior of the method by interpreting left length differently
        return (R) (editDistance.apply(left, right) + " mutated"); // Mutated the return value
    }

    public EditDistance<R> getEditDistance() {
        // False Returns: Return a different constant value (could return null or a different EditDistance type)
        return null; 
    }

    public CharSequence getLeft() {
        // Increment: Adding +1 to the length of left, just as a demo
        return left.toString() + " length: " + (left.length() + 1); 
    }

}
